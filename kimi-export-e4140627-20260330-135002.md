// src/components/pdf-viewer/PDFViewer.tsx
import { useEffect, useRef, useState } from 'react';
import * as pdfjsLib from 'pdfjs-dist';
import { LayerManager } from './layers/LayerManager';

// 使用 pdfium 后端获取 PDF (更快，支持更多格式)
import { invoke } from '@tauri-apps/api/tauri';

pdfjsLib.GlobalWorkerOptions.workerSrc = '/pdf.worker.js';

interface PDFViewerProps {
  filePath: string;
}

export function PDFViewer({ filePath }: PDFViewerProps) {
  const containerRef = useRef<HTMLDivElement>(null);
  const [pdfDoc, setPdfDoc] = useState<pdfjsLib.PDFDocumentProxy | null>(null);
  const [currentPage, setCurrentPage] = useState(1);
  const [scale, setScale] = useState(1.5);
  const [pages, setPages] = useState<pdfjsLib.PDFPageProxy[]>([]);

  // 加载 PDF
  useEffect(() => {
    loadPDF();
  }, [filePath]);

  const loadPDF = async () => {
    // 方案1: 通过 Rust 读取文件（推荐，可以处理大文件）
    const pdfBytes = await invoke<number[]>('read_pdf_file', { path: filePath });
    const uint8Array = new Uint8Array(pdfBytes);
    const pdf = await pdfjsLib.getDocument({ data: uint8Array }).promise;
    
    setPdfDoc(pdf);
    
    // 预加载所有页面
    const pagePromises = [];
    for (let i = 1; i <= pdf.numPages; i++) {
      pagePromises.push(pdf.getPage(i));
    }
    const loadedPages = await Promise.all(pagePromises);
    setPages(loadedPages);
  };

  return (
    <div className="pdf-viewer" ref={containerRef}>
      {/* 工具栏 */}
      <Toolbar 
        currentPage={currentPage} 
        totalPages={pdfDoc?.numPages || 0}
        scale={scale}
        onPageChange={setCurrentPage}
        onScaleChange={setScale}
      />
      
      {/* PDF 页面容器 */}
      <div className="pages-container">
        {pages.map((page, index) => (
          <PageComponent 
            key={index}
            page={page}
            pageNumber={index + 1}
            scale={scale}
            filePath={filePath}
          />
        ))}
      </div>
    </div>
  );
}

// 单页组件
function PageComponent({ page, pageNumber, scale, filePath }: PageProps) {
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const [viewport, setViewport] = useState<any>(null);

  useEffect(() => {
    renderPage();
  }, [page, scale]);

  const renderPage = async () => {
    const canvas = canvasRef.current;
    if (!canvas) return;

    const viewport = page.getViewport({ scale });
    setViewport(viewport);
    
    canvas.width = viewport.width;
    canvas.height = viewport.height;
    
    const ctx = canvas.getContext('2d')!;
    await page.render({
      canvasContext: ctx,
      viewport: viewport
    }).promise;
  };

  return (
    <div className="page-wrapper" data-page-number={pageNumber}>
      {/* PDF 渲染层 */}
      <canvas ref={canvasRef} className="pdf-canvas" />
      
      {/* 标注层 - 叠加在 PDF 之上 */}
      <LayerManager
        filePath={filePath}
        pageNumber={pageNumber}
        width={viewport?.width || 0}
        height={viewport?.height || 0}
        scale={scale}
      />
    </div>
  );
}

export function LayerManager({ filePath, pageNumber, width, height, scale }: LayerManagerProps) {
  const containerRef = useRef<HTMLDivElement>(null);

fn apply_eye_protection(texture: &mut Texture, mode: ColorMode) {
    let matrix = match mode {
        ColorMode::Normal => ColorMatrix::identity(),
        ColorMode::Dark => ColorMatrix::new([
            0.9, 0.0, 0.0, 0.0,  // R
            0.0, 0.9, 0.0, 0.0,  // G
            0.0, 0.0, 0.9, 0.0,  // B
            0.0, 0.0, 0.0, 1.0,  // A

        ColorMode::Sepia => ColorMatrix::sepia(0.4),
    };
    texture.apply_color_matrix(matrix);
}

```rust
// 启动时检查文件变化
fn sync_library() {
    // 1. 扫描 library/ 目录，发现新 PDF
    let disk_pdfs = scan_pdf_files(&config.library_path);
    
    // 2. 对比数据库记录
    let db_pdfs = db.query_all_pdfs();
    
    // 3. 处理差异
    for pdf in disk_pdfs.difference(&db_pdfs) {
        // 新文件：提取元数据，加入数据库
        db.insert_pdf(pdf);
        generate_thumbnail(pdf);
    }
    
    // 4. 监听文件变化（fswatch）
    start_watching(&config.library_path);
}
```
